PolygonGenerators.createPolygon = function(nPoints, mode, frame) {
  var polygon = new Polygon();

  switch(mode) {
    case 0: //random
      for(var i = 0; i < nPoints; i++) {
        polygon.push(new Point(frame.x + frame.width * Math.random(), frame.y + frame.height * Math.random()));
      }
      break;
    case 1: //circle
      break;
  }

  return polygon;
};
